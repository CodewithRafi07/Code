//Problem_3: Calculate the Factorial of a Number

#include<iostream>
using namespace std;

int main() {
    int number, i = 1;
    int Product=1;

    cout << "Enter a Number ";
    cin >> number;

    while (i <= number)
    {
        Product = Product * i;

}



//Problem_4: Print Fibonacci Series

#include<iostream>
using namespace std;

int main (){
    int n, a, b, next = 0;

    cout << "Enter the first TWO numbers (Using one space) " ;
    cin >> a >> b;

    cout << "Enter the Limit or Last Number ";
    cin >> n;

    cout << "Fibonacci Series up to " << n << " is ";

    while (a <= n) {
        cout << a << " ";
        next = a + b;
        a = b;
        b = next;
    }

    return 0;
}



//Problem_5: Reverse a Number

#include<iostream>
using namespace std;

int main() {
    int number, tnum, rev=0;

    cout << "Enter the Number: ";
    cin >> number;

    while (number!= 0){
        tnum = number % 10;
        rev = rev * 10 + tnum;
        number /= 10;
    }

    cout << "The reversed Number is " << rev << endl;
    return 0;
}



//Problem_6: Check whether a number is Palindrome

#include <iostream>
using namespace std;

int main() {
    int num, orgnl, rev = 0;

    cout << "Enter a number: ";
    cin >> num;

    orgnl = num;

    while (num != 0) {
        int digit = num % 10;
        rev = rev * 10 + digit;
        num /= 10;
    }

    if (orgnl == rev) {
        cout << orgnl << " is a palindrome" << endl;
    } else {
        cout << orgnl << " is not a palindrome" << endl;
    }

    return 0;
}

//Problem_8: Find GCD of Two Numbers

#include <iostream>
using namespace std;

int main() {
    int a, b;

    cout << "Enter first number: ";
    cin >> a;

    cout << "Enter second number: ";
    cin >> b;

    int gcd;

    for (int i = 1; i <= a && i <= b; i++) {
        if (a % i == 0 && b % i == 0) {
            gcd = i;
        }
    }
    cout << "The GCD of " << a << " and " << b << " is " << gcd << endl;

    return 0;
}

//Problem_10: Multiplication table of a Number

using namespace std;

    int num;

    cout << "Multiplication table of " << num << ":\n";
